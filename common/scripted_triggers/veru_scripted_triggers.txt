						has_overlord = from

				}

			use_bypasses = yes
			max_steps = 1
		}
	}

ver_blood_reqs = {
	# basic_vassal_reqs = yes
	is_gestalt = no
	is_tributary = no
	is_protectorate = no
	$OL$ = {
		is_gestalt = no
		OR = {
			is_death_cult_empire = yes
			has_ver_blood_civic = yes
		}
	}
	OR = { ## Limiter
		$OL$ = { NOT = { any_subject = { is_ver_blood = yes } } }
		AND = {
			$OL$ = {
				count_subject = {
					count = 1
					limit = { is_ver_blood = yes }
				}
			}
			is_ver_blood = yes
		}	
	}
}
